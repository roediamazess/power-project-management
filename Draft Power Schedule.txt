  <script>
    let currentUser = null;
    let allUsers = [];
    let allSchedules = [];
    let allBatches = [];
    let pendingAction = null;

    const dataHandler = {
      onDataChanged(data) {
        separateData(data);
        if (currentUser) {
          renderDashboard();
        }
      }
    };

    // Initialize SDK
    (async () => {
      const result = await window.dataSdk.init(dataHandler);
      if (!result.isOk) {
        showToast('Error initializing data');
      }

      window.elementSdk?.init({
        defaultConfig: { app_title: 'Submission Schedule System' },
        onConfigChange: () => {},
        mapToCapabilities: () => ({ recolorables: [], borderables: [], fontEditable: undefined, fontSizeable: undefined }),
        mapToEditPanelValues: (config) => new Map([['app_title', config.app_title]])
      });

      setupEventListeners();
    })();

    function separateData(data) {
      allUsers = data.filter(d => d.type === 'user');
      allSchedules = data.filter(d => d.type === 'schedule');
      allBatches = data.filter(d => d.type === 'batch');
    }

    function setupEventListeners() {
      document.getElementById('themeToggle').addEventListener('click', toggleTheme);
      
      const savedTheme = localStorage.getItem('theme') || 'light';
      if (savedTheme === 'dark') {
        document.body.classList.add('dark-mode');
        document.getElementById('themeToggle').textContent = '☀️';
      }

      document.getElementById('loginForm').addEventListener('submit', handleLogin);
      document.getElementById('registerForm').addEventListener('submit', handleRegister);
      document.getElementById('toggleRegister').addEventListener('click', toggleScreen('registerScreen'));
      document.getElementById('toggleLogin').addEventListener('click', toggleScreen('loginScreen'));
      document.getElementById('logoutBtn').addEventListener('click', handleLogout);

      document.getElementById('createScheduleBtn').addEventListener('click', () => {
        document.getElementById('createScheduleModal').classList.remove('hidden');
      });
      document.getElementById('cancelCreateBtn').addEventListener('click', () => {
        document.getElementById('createScheduleModal').classList.add('hidden');
      });
      document.getElementById('createScheduleForm').addEventListener('submit', handleCreateSchedule);

      document.getElementById('createBatchBtn').addEventListener('click', showCreateBatchModal);
      document.getElementById('cancelBatchBtn').addEventListener('click', () => {
        document.getElementById('createBatchModal').classList.add('hidden');
      });
      document.getElementById('createBatchForm').addEventListener('submit', handleCreateBatch);

      document.getElementById('manageTiersBtn').addEventListener('click', showManageTiersModal);
      document.getElementById('closeTiersBtn').addEventListener('click', () => {
        document.getElementById('manageTiersModal').classList.add('hidden');
      });

      document.getElementById('cancelActionBtn').addEventListener('click', () => {
        document.getElementById('actionModal').classList.add('hidden');
        pendingAction = null;
      });
      document.getElementById('confirmActionBtn').addEventListener('click', handleConfirmAction);
    }

    function toggleScreen(screenId) {
      return () => {
        document.getElementById('loginScreen').classList.add('hidden');
        document.getElementById('registerScreen').classList.add('hidden');
        document.getElementById(screenId).classList.remove('hidden');
      };
    }

    function toggleTheme() {
      const body = document.body;
      const themeBtn = document.getElementById('themeToggle');
      
      body.classList.toggle('dark-mode');
      
      if (body.classList.contains('dark-mode')) {
        localStorage.setItem('theme', 'dark');
        themeBtn.textContent = '☀️';
      } else {
        localStorage.setItem('theme', 'light');
        themeBtn.textContent = '🌙';
      }
    }

    async function handleLogin(e) {
      e.preventDefault();
      const username = document.getElementById('loginUsername').value;
      const password = document.getElementById('loginPassword').value;

      let user = allUsers.find(u => u.username === username && u.password === password);
      
      if (!user) {
        if (username === 'admin' && password === 'admin123') {
          user = {
            type: 'user',
            username: 'admin',
            password: 'admin123',
            role: 'admin',
            tier: 'admin',
            point: 0,
            schedule_id: '',
            schedule_name: '',
            description: '',
            start_date: '',
            end_date: '',
            batch_id: '',
            batch_name: '',
            point_min: 0,
            point_max: 0,
            status: '',
            picked_by: '',
            released_at: '',
            created_by: '',
            created_at: ''
          };
        } else if (username === 'user1' && password === 'pass123') {
          user = {
            type: 'user',
            username: 'user1',
            password: 'pass123',
            role: 'team',
            tier: 'tier_2',
            point: 2,
            schedule_id: '',
            schedule_name: '',
            description: '',
            start_date: '',
            end_date: '',
            batch_id: '',
            batch_name: '',
            point_min: 0,
            point_max: 0,
            status: '',
            picked_by: '',
            released_at: '',
            created_by: '',
            created_at: ''
          };
        }
      }

      if (user) {
        currentUser = user;
        showLoginUI();
      } else {
        showToast('Username atau password salah');
      }
    }

    async function handleRegister(e) {
      e.preventDefault();
      const username = document.getElementById('regUsername').value;
      const password = document.getElementById('regPassword').value;
      const role = document.getElementById('regRole').value;

      if (allUsers.some(u => u.username === username)) {
        showToast('Username sudah terdaftar');
        return;
      }

      const result = await window.dataSdk.create({
        type: 'user',
        username,
        password,
        role,
        tier: role === 'admin' ? 'admin' : 'new_born',
        point: role === 'admin' ? 0 : 1,
        schedule_id: '',
        schedule_name: '',
        description: '',
        start_date: '',
        end_date: '',
        batch_id: '',
        batch_name: '',
        point_min: 0,
        point_max: 0,
        status: '',
        picked_by: '',
        released_at: '',
        created_by: '',
        created_at: ''
      });

      if (result.isOk) {
        showToast('Registrasi berhasil! Silakan login');
        document.getElementById('toggleLogin').click();
      }
    }

    function handleLogout() {
      currentUser = null;

    function renderGroupedAvailableSchedules(schedules, type) {
      const scheduleOrder = ['Middle', 'Day', 'Duty', 'Saturday', 'Sunday', 'Public Holiday'];
      let html = '<div class="mb-8">';

      scheduleOrder.forEach(scheduleName => {

    function formatDate(dateString) {
      const date = new Date(dateString + 'T00:00:00');
      const options = { day: '2-digit', month: 'short', year: '2-digit' };
      return date.toLocaleDateString('en-GB', options).replace(/ /g, ' ');
    }

    async function handleConfirmAction() {
      if (!pendingAction) return;

      const sched = allSchedules.find(s => s.__backendId === pendingAction.schedId);
      if (!sched) return;

      let updated = { ...sched };
      let message = '';

      if (pendingAction.action === 'pickup') {
        // Check if schedule is in a batch
        if (sched.batch_id) {
          const batchSchedules = allSchedules.filter(s => s.batch_id === sched.batch_id && s.status === 'picked');
          
          let totalPointsCurrently = 0;
          let pickedUsers = [];

          for (const schedInBatch of batchSchedules) {
            const pickedByUser = allUsers.find(u => u.username === schedInBatch.picked_by);
            if (pickedByUser) {
              totalPointsCurrently += getTierPoints(pickedByUser.tier);
              pickedUsers.push(`${pickedByUser.username} (${getTierPoints(pickedByUser.tier)} pts)`);
            }
          }

          // Add this user's points
          const totalPointsWithUser = totalPointsCurrently + getTierPoints(currentUser.tier);

          // Check if total points meet requirement
          if (totalPointsWithUser < sched.point_min || totalPointsWithUser > sched.point_max) {
            let requiredMsg = `❌ Batch "${sched.batch_name}" requirement tidak terpenuhi.\n\n`;
            requiredMsg += `Required: ${sched.point_min}-${sched.point_max} points\n`;
            requiredMsg += `Current total: ${totalPointsCurrently} points\n`;
            requiredMsg += `+ Your tier: ${getTierName(currentUser.tier)} (${getTierPoints(currentUser.tier)} pts)\n`;
            requiredMsg += `= Total would be: ${totalPointsWithUser} points\n`;
            if (pickedUsers.length > 0) {
              requiredMsg += `Already picked: ${pickedUsers.join(', ')}\n`;
            }
            showToast(requiredMsg);
            document.getElementById('actionModal').classList.add('hidden');
            pendingAction = null;
            return;
          }
        }

        updated.status = 'picked';
        updated.picked_by = currentUser.username;
        message = 'Schedule berhasil di-pick up';
      } else if (pendingAction.action === 'reopen') {
        updated.status = 'available';
        updated.picked_by = '';
        message = 'Schedule berhasil di-reopen';
      } else if (pendingAction.action === 'release') {
        updated.status = 'released';
        updated.released_at = new Date().toISOString();
        message = 'Schedule berhasil di-release';
      }

      const result = await window.dataSdk.update(updated);
      if (result.isOk) {
        showToast(message);
        document.getElementById('actionModal').classList.add('hidden');
        pendingAction = null;
      }
    }

    async function handleCreateBatch(e) {
      e.preventDefault();
      
      const batchName = document.getElementById('batchName').value;
      const pointMin = parseInt(document.getElementById('batchPointMin').value);
      const pointMax = parseInt(document.getElementById('batchPointMax').value);

      const selectedCheckboxes = document.querySelectorAll('.sched-checkbox:checked');
      if (selectedCheckboxes.length === 0) {
        showToast('Pilih minimal 1 schedule');
        return;
      }

      const batchId = 'batch_' + Date.now();

      for (const checkbox of selectedCheckboxes) {
        const schedId = checkbox.value;
        const sched = allSchedules.find(s => s.__backendId === schedId);
        
        const updated = { ...sched };
        updated.batch_id = batchId;
        updated.batch_name = batchName;
        updated.point_min = pointMin;
        updated.point_max = pointMax;

        const result = await window.dataSdk.update(updated);
        if (!result.isOk) {
          showToast('Error updating schedule');
          return;
        }
      }

      document.getElementById('createBatchModal').classList.add('hidden');
      document.getElementById('createBatchForm').reset();
      showToast(`✅ Batch "${batchName}" (${pointMin}-${pointMax} pts) berhasil dibuat dengan ${selectedCheckboxes.length} schedule`);
    }
