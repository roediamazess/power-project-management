    function isOverlapping(startA, endA, startB, endB) {

    function pushNotification(target, text) {
      notifications.unshift({ id: generateId('notif'), target: target, text: text, read: false, time: new Date() });
      if(currentUser) renderNotifications();
    }

    function renderNotifications() {
      if(!currentUser) return;
      const myNotifs = notifications.filter(n => n.target === currentUser.username || (currentUser.role === 'admin' && n.target === 'all_admin'));
      const unreadCount = myNotifs.filter(n => !n.read).length;
      const badge = document.getElementById('notifBadge');
      if(unreadCount > 0) {
        badge.textContent = unreadCount > 9 ? '9+' : unreadCount;
        badge.style.display = 'flex';
      } else {
        badge.style.display = 'none';
      }
      const listContainer = document.getElementById('notifList');
      if(myNotifs.length === 0) {
         listContainer.innerHTML = '<p class="text-sm text-gray-500 p-4 text-center italic">Belum ada notifikasi.</p>';
      } else {
         listContainer.innerHTML = myNotifs.map(n => `
           <div class="p-3 border-b border-gray-50 dark:border-slate-700/50 hover:bg-gray-50 dark:hover:bg-slate-700 transition ${n.read ? 'opacity-60' : 'bg-blue-50/30 dark:bg-blue-900/10'}">
             <p class="text-sm text-gray-800 dark:text-gray-200 ${n.read ? '' : 'font-semibold'}">${n.text}</p>
             <span class="text-[10px] text-gray-400 mt-1 block">${n.time.toLocaleTimeString([], {hour: '2-digit', minute:'2-digit'})} - ${n.time.toLocaleDateString()}</span>
           </div>
         `).join('');
      }
    }

    function populatePeriodSelector(selectId = null) {
      const selector = document.getElementById('periodSelector');
      if (!selector) return;
      selector.innerHTML = '';
      if(allPeriods.length === 0){
        selector.innerHTML = '<option value="">-- Belum ada periode --</option>';
        return;
      }
      allPeriods.forEach(p => {
        const opt = document.createElement('option');
        opt.value = p.id;
        opt.textContent = p.name;
        selector.appendChild(opt);
      });
      if(selectId) selector.value = selectId;
    }

    function showScheduleModal(mode, schedId = null) {
      const modalTitle = document.getElementById('scheduleModalTitle');
      const form = document.getElementById('scheduleForm');
      const supportContainer = document.getElementById('schedSupportContainer');
      const idInput = document.getElementById('schedEditId');
      
      if(form) form.reset();

      if (mode === 'create') {
        if(modalTitle) modalTitle.textContent = "Create New Schedule";
        if(supportContainer) supportContainer.classList.remove('hidden');
        if(idInput) idInput.value = '';
      } else if (mode === 'edit' && schedId) {
        if(modalTitle) modalTitle.textContent = "Edit Schedule";
        if(supportContainer) supportContainer.classList.add('hidden');
        if(idInput) idInput.value = schedId;

        const sched = allSchedules.find(s => s.__backendId === schedId);
        if (sched) {
          const nameInput = document.getElementById('schedName');
          const descInput = document.getElementById('schedDesc');
          const startInput = document.getElementById('schedStart');
          const endInput = document.getElementById('schedEnd');
          
          if(nameInput) nameInput.value = sched.schedule_name;
          if(descInput) descInput.value = sched.description;
          if(startInput) startInput.value = sched.start_date;
          if(endInput) endInput.value = sched.end_date;
        }
      }
      
      const modal = document.getElementById('scheduleModal');
      if(modal) modal.classList.remove('hidden');
    }

    function handleSaveSchedule(e) {
      e.preventDefault();
      const schedId = document.getElementById('schedEditId').value;
      const name = document.getElementById('schedName').value;
      const desc = document.getElementById('schedDesc').value;
      const start = document.getElementById('schedStart').value;
      const end = document.getElementById('schedEnd').value;

      if (new Date(start) > new Date(end)) {
         showToast('❌ Start Date tidak boleh melewati End Date');
         return;
      }

      if (schedId) {
        const schedIndex = allSchedules.findIndex(s => s.__backendId === schedId);
        if (schedIndex > -1) {
          allSchedules[schedIndex].schedule_name = name;
          allSchedules[schedIndex].description = desc;
          allSchedules[schedIndex].start_date = start;
          allSchedules[schedIndex].end_date = end;
          showToast('✅ Schedule berhasil diperbarui');
        }
      } else {
        const supportNeeded = parseInt(document.getElementById('schedSupport').value) || 1;
        for (let i = 0; i < supportNeeded; i++) {
          allSchedules.push({
            __backendId: generateId('sched'),
            type: 'schedule', schedule_name: name, description: desc,
            start_date: start, end_date: end, status: 'available',
            batch_id: '', batch_name: '', point_min: 0, point_max: 0,
            picked_by: '', released_at: '', created_by: currentUser.username
          });
        }
        showToast(`✅ ${supportNeeded} schedule(s) berhasil dibuat`);
      }

      document.getElementById('scheduleModal').classList.add('hidden');
      renderDashboard();
    }

    function handleCreateBatch(e) {
      e.preventDefault();
      const batchName = document.getElementById('batchName').value;
      const pointMin = parseInt(document.getElementById('batchPointMin').value);
      const pointMax = parseInt(document.getElementById('batchPointMax').value);
      
      if (pointMin > pointMax) {
        showToast('❌ Min point tidak boleh lebih besar dari max point');
        return;
      }

      const selectedCheckboxes = document.querySelectorAll('.sched-checkbox:checked');
      if (selectedCheckboxes.length === 0) {
        showToast('❌ Pilih minimal 1 schedule');
        return;
      }

      const batchId = generateId('batch');
      selectedCheckboxes.forEach(checkbox => {
        const schedIndex = allSchedules.findIndex(s => s.__backendId === checkbox.value);
        if (schedIndex > -1) {
          allSchedules[schedIndex].batch_id = batchId;
          allSchedules[schedIndex].batch_name = batchName;
          allSchedules[schedIndex].point_min = pointMin;
          allSchedules[schedIndex].point_max = pointMax;
        }
      });

      document.getElementById('createBatchModal').classList.add('hidden');
      document.getElementById('createBatchForm').reset();
      renderDashboard();
      showToast(`✅ Batch "${batchName}" berhasil dibuat dengan ${selectedCheckboxes.length} jadwal`);
    }

    function isHoliday(dateObj) {
      const yyyy = dateObj.getFullYear();
      const mm = String(dateObj.getMonth() + 1).padStart(2, '0');
      const dd = String(dateObj.getDate()).padStart(2, '0');
      const dateStr = `${yyyy}-${mm}-${dd}`;
      return allHolidays.some(h => h.date === dateStr);
    }

      } else {
        const sortedHolidays = [...allHolidays].sort((a,b) => new Date(a.date) - new Date(b.date));

            <button class="text-red-500 hover:text-red-700 font-bold text-lg px-2 del-holiday-btn" data-id="${h.id}">×</button>

        `).join('');
        document.querySelectorAll('.del-holiday-btn').forEach(btn => {

    function showActionModal(title, message, action, schedId) {
      const titleEl = document.getElementById('actionModalTitle');
      const messageEl = document.getElementById('actionModalMessage');
      if (titleEl) titleEl.textContent = title;
      if (messageEl) messageEl.textContent = message;
      document.getElementById('actionModal')?.classList.remove('hidden');
      pendingAction = { action, schedId };
    }

    function handleConfirmAction() {
      if (!pendingAction) return;
      const schedIndex = allSchedules.findIndex(s => s.__backendId === pendingAction.schedId);
      if (schedIndex === -1) return;
      const sched = allSchedules[schedIndex];

      if (pendingAction.action === 'delete') {
        allSchedules.splice(schedIndex, 1);
        showToast('🗑️ Schedule dihapus');
      } else if (pendingAction.action === 'pickup') {
        const myAgendas = allAgendas.filter(a => a.username === currentUser.username);
        const hasAgendaConflict = myAgendas.some(a => isOverlapping(sched.start_date, sched.end_date, a.start_date, a.end_date));
        if (hasAgendaConflict) { showToast('❌ Bentrok! Anda memiliki agenda/cuti lain.'); document.getElementById('actionModal').classList.add('hidden'); return; }

        const myScheds = allSchedules.filter(s => s.picked_by === currentUser.username && (s.status === 'picked' || s.status === 'released'));
        const hasSchedConflict = myScheds.some(s => isOverlapping(sched.start_date, sched.end_date, s.start_date, s.end_date));
        if (hasSchedConflict) { showToast('❌ Bentrok! Anda sudah memiliki jadwal di rentang ini.'); document.getElementById('actionModal').classList.add('hidden'); return; }

        allSchedules[schedIndex].status = 'picked';
        allSchedules[schedIndex].picked_by = currentUser.username;
        showToast('✅ Schedule berhasil di-pick up');
        pushNotification('all_admin', `${currentUser.username} mem-pickup jadwal ${sched.schedule_name}`);

      } else if (pendingAction.action === 'reopen') {
        allSchedules[schedIndex].status = 'available';
        allSchedules[schedIndex].picked_by = '';
        showToast('🔄 Schedule kembali tersedia');
      } else if (pendingAction.action === 'release') {
        allSchedules[schedIndex].status = 'released';
        allSchedules[schedIndex].released_at = new Date().toISOString();
        showToast('📤 Schedule berhasil di-publish ke Jobsheet');
        pushNotification(sched.picked_by, `Jadwal ${sched.schedule_name} baru saja di-publish ke Jobsheet Anda`);
      }

      document.getElementById('actionModal')?.classList.add('hidden');
      pendingAction = null;
      renderDashboard();
    }

    function toggleScreen(screenId) {

        const targetScreen = document.getElementById(screenId);

    function checkTheme() {
      const savedTheme = localStorage.getItem('theme') || 'light';
      if (savedTheme === 'dark') {
        document.documentElement.classList.add('dark');
        document.body.classList.add('dark-mode');
        const themeToggle = document.getElementById('themeToggle');
        if (themeToggle) themeToggle.textContent = '☀️';
      } else {
        document.documentElement.classList.remove('dark');
        document.body.classList.remove('dark-mode');
      }
    }

    function toggleTheme() {
      const isDark = document.documentElement.classList.toggle('dark');
      document.body.classList.toggle('dark-mode');
      const themeToggle = document.getElementById('themeToggle');
      if (isDark) {
        localStorage.setItem('theme', 'dark');
        if (themeToggle) themeToggle.textContent = '☀️';
      } else {
        localStorage.setItem('theme', 'light');
        if (themeToggle) themeToggle.textContent = '🌙';
      }
      if(currentTab === 'analytics') renderAnalytics();
    }

    function showToast(message) {
      const toast = document.getElementById('toast');
      const toastMsg = document.getElementById('toastMessage');
      if (toast && toastMsg) {
          toastMsg.textContent = message;
          toast.classList.remove('hidden');
          setTimeout(() => toast.classList.add('hidden'), 4000);
      }
    }

        if (currentUser.role === 'admin') { btnType = 'admin'; }

           if (sched.status === 'available') btnType = 'team';
           else if (sched.status === 'picked' && sched.picked_by === currentUser.username) btnType = 'team-owned-picked';
           else if (sched.status === 'released' && sched.picked_by === currentUser.username) btnType = 'team-owned-released';
           else if (sched.picked_by !== currentUser.username) btnType = 'team-others';
        }

        html += `
          <div class="bg-white dark:bg-slate-900 rounded-xl shadow-md p-6 card-hover flex flex-col h-full border border-gray-200 dark:border-slate-700">
            <div class="flex justify-between items-start mb-4">

    // --- GRID VIEW (Jobsheet) Logic & Drag Edit ---
    function renderGridView() {
      const container = document.getElementById('gridContainer');
      const periodSelector = document.getElementById('periodSelector');
      const periodId = periodSelector ? periodSelector.value : null;
      const searchInput = document.getElementById('searchGridUser');
      const searchTerm = searchInput ? searchInput.value.toLowerCase() : '';
      const period = allPeriods.find(p => p.id === periodId);

      if (!period) {
        if(container) container.innerHTML = '<p class="text-gray-500 italic p-4 text-center">Pilih periode untuk melihat jadwal...</p>';
        return;
      }

      const dates = [];
      const startParts = period.start.split('-');
      const endParts = period.end.split('-');
      let currDate = new Date(startParts[0], startParts[1] - 1, startParts[2]);
      const endDate = new Date(endParts[0], endParts[1] - 1, endParts[2]);

      while(currDate <= endDate) {
        dates.push(new Date(currDate));
        currDate.setDate(currDate.getDate() + 1);
      }

      let tableHtml = `<div class="mb-4 text-center font-bold text-xl uppercase text-gray-800 dark:text-white border-b-2 border-blue-500 pb-2 inline-block">SUPPORT IN CHARGE SCHEDULE PERIODE ${period.name.toUpperCase()}</div>
        <table class="grid-table bg-white dark:bg-slate-800" id="jobsheetTable">
          <thead>
            <tr>
              <th rowspan="2" class="sticky-col border-r-2 border-gray-400 min-w-[150px] uppercase text-sm tracking-wider">PIC</th>`;
      
      dates.forEach(d => {
        const dayOfWeek = d.getDay();
        const isHol = isHoliday(d);
        let classStr = "text-gray-800 dark:text-gray-200";
        if (isHol) classStr = "bg-holiday";
        else if (dayOfWeek === 6) classStr = "bg-saturday";
        else if (dayOfWeek === 0) classStr = "bg-sunday";
        tableHtml += `<th class="${classStr}">${d.toLocaleDateString('en-GB', { month: 'short' })}</th>`;
      });
      tableHtml += `</tr><tr>`;
      
      dates.forEach(d => {
        const dayOfWeek = d.getDay();
        const isHol = isHoliday(d);
        let classStr = "text-gray-800 dark:text-gray-200 font-bold";
        if (isHol) classStr = "bg-holiday font-bold";
        else if (dayOfWeek === 6) classStr = "bg-saturday font-bold";
        else if (dayOfWeek === 0) classStr = "bg-sunday font-bold";
        tableHtml += `<th class="${classStr}">${d.getDate().toString().padStart(2, '0')}</th>`;
      });

      tableHtml += `</tr></thead><tbody>`;

      let teamUsers = allUsers.filter(u => u.role === 'team').sort((a,b) => a.username.localeCompare(b.username));
      if (searchTerm) teamUsers = teamUsers.filter(u => u.username.toLowerCase().includes(searchTerm));

      if (teamUsers.length === 0) {
        tableHtml += `<tr><td colspan="${dates.length + 1}" class="p-4 text-center italic text-gray-500">Tidak ada user ditemukan.</td></tr>`;
      }

      teamUsers.forEach(user => {
        tableHtml += `<tr>
          <td class="sticky-col border-r-2 border-gray-400 text-left font-medium text-gray-700 dark:text-gray-300">${user.username}</td>`;
          
        dates.forEach(d => {
            const dayOfWeek = d.getDay();
            const isHol = isHoliday(d);
            let tdClass = "bg-gray-50 dark:bg-slate-800/50 transition-colors text-center font-bold align-middle border";
            
            if (isHol) tdClass = "bg-gray-300 dark:bg-slate-600 border-gray-400 dark:border-slate-500 opacity-90 text-center font-bold align-middle text-gray-800 dark:text-white border"; 
            else if(dayOfWeek === 6) tdClass = "bg-saturday opacity-90 text-center font-bold align-middle border"; 
            else if(dayOfWeek === 0) tdClass = "bg-sunday opacity-90 text-center font-bold align-middle border"; 
            
            const yyyy = d.getFullYear();
            const mm = String(d.getMonth() + 1).padStart(2, '0');
            const dd = String(d.getDate()).padStart(2, '0');
            const dateStr = `${yyyy}-${mm}-${dd}`;

            const userAgendas = allAgendas.filter(a => a.username === user.username && dateStr >= a.start_date && dateStr <= a.end_date);
            let cellTextArray = [];
            let isEditableInline = true;

            if (userAgendas.length > 0) {
                if (userAgendas[0].desc !== 'Inline Input') isEditableInline = false;
                const agendaType = userAgendas[0].type;
                let bgClass = "bg-yellow-200 dark:bg-yellow-800 border-yellow-300 dark:border-yellow-700 text-yellow-800 dark:text-yellow-200";

                if (agendaType.startsWith('I.')) bgClass = "bg-indigo-200 dark:bg-indigo-800 border-indigo-300 dark:border-indigo-700 text-indigo-800 dark:text-indigo-200";
                else if (agendaType.startsWith('M.')) bgClass = "bg-emerald-200 dark:bg-emerald-800 border-emerald-300 dark:border-emerald-700 text-emerald-800 dark:text-emerald-200";
                else if (['MS', 'OKR', 'TM'].includes(agendaType)) bgClass = "bg-purple-200 dark:bg-purple-800 border-purple-300 dark:border-purple-700 text-purple-800 dark:text-purple-200";

                tdClass = `${bgClass} opacity-90 text-center align-middle border`;
                cellTextArray.push(`<span class="font-extrabold text-xs tracking-wide">${agendaType}</span>`);
            } else {
                const userScheds = allSchedules.filter(s => s.status === 'released' && s.picked_by === user.username && dateStr >= s.start_date && dateStr <= s.end_date);
                if (userScheds.length > 0 && userScheds.some(s => s.description !== 'Inline Input')) {
                    isEditableInline = false;
                }

                userScheds.forEach(sched => {
                    if (sched.schedule_name === 'Day') cellTextArray.push('<span class="text-[#c2410c] dark:text-orange-400 font-extrabold tracking-wide">D</span>');
                    else if (sched.schedule_name === 'Middle') cellTextArray.push('<span class="text-[#4fd1c5] dark:text-teal-300 font-extrabold tracking-wide">MD</span>');
                    else if (sched.schedule_name === 'Duty') {
                        let skipSunday = false;
                        if (dayOfWeek === 0) { 
                            const prevDay = new Date(d); prevDay.setDate(prevDay.getDate() - 1);
                            const pDateStr = `${prevDay.getFullYear()}-${String(prevDay.getMonth() + 1).padStart(2, '0')}-${String(prevDay.getDate()).padStart(2, '0')}`;
                            if (sched.start_date <= pDateStr && sched.end_date >= pDateStr) skipSunday = true; 
                        }
                        if (!skipSunday) cellTextArray.push('<span class="text-blue-700 dark:text-blue-400 font-extrabold tracking-wide">DT</span>');
                    } 
                    else if (sched.schedule_name === 'Public Holiday') cellTextArray.push('PH');
                    else if (sched.schedule_name === 'Saturday') cellTextArray.push('Sat');
                    else if (sched.schedule_name === 'Sunday') cellTextArray.push('Sun');
                    else cellTextArray.push(sched.schedule_name.substring(0,2).toUpperCase());
                });
            }
            
            let editableAttr = '';
            if(currentUser.role === 'admin') {
                if (isEditableInline) {
                    tdClass += " editable-cell cursor-cell select-none relative";
                    editableAttr = `data-date="${dateStr}" data-user="${user.username}" title="Klik & geser untuk edit"`;
                } else {
                    tdClass += " cursor-not-allowed opacity-80 select-none";
                    editableAttr = `title="Dibuat dari Board. Edit via Board."`;
                }
            }

            tableHtml += `<td class="${tdClass}" ${editableAttr}>${cellTextArray.join(' / ')}</td>`;
        });
        tableHtml += `</tr>`;
      });

      tableHtml += `</tbody></table>`;
      if(container) {
          container.innerHTML = tableHtml;
          attachGridDragListeners();
      }
    }

    // --- Drag & Drop Selection Logic ---
    function attachGridDragListeners() {
        const table = document.getElementById('jobsheetTable');
        if (!table) return;

        // Mouse Events
        table.onmousedown = handleDragStart;
        table.onmouseover = handleDragMove;
        window.removeEventListener('mouseup', handleDragEnd); // prevent duplicates
        window.addEventListener('mouseup', handleDragEnd);

        // Touch Events
        table.addEventListener('touchstart', handleTouchStart, {passive: false});
        table.addEventListener('touchmove', handleTouchMove, {passive: false});
        window.addEventListener('touchend', handleDragEnd);
        
        // Double click inline edit
        table.ondblclick = handleInlineEdit;
    }

    function handleDragStart(e) {
        if (currentUser?.role !== 'admin' || e.button !== 0) return; // Only left click & admin
        const td = e.target.closest('td.editable-cell');
        if (!td) return;

        isDragging = true;
        dragUser = td.dataset.user;
        dragStartD = td.dataset.date;
        dragEndD = dragStartD;
        
        highlightGridRange(dragUser, dragStartD, dragEndD);
    }

    function handleDragMove(e) {
        if (!isDragging) return;
        const td = e.target.closest('td.editable-cell');
        if (!td || td.dataset.user !== dragUser) return; // Must be same row (user)
        
        dragEndD = td.dataset.date;
        highlightGridRange(dragUser, dragStartD, dragEndD);
    }

    function handleTouchStart(e) {
        if (currentUser?.role !== 'admin') return;
        const touch = e.touches[0];
        const el = document.elementFromPoint(touch.clientX, touch.clientY);
        if(!el) return;
        const td = el.closest('td.editable-cell');
        if (!td) return;

        isDragging = true;
        dragUser = td.dataset.user;
        dragStartD = td.dataset.date;
        dragEndD = dragStartD;
        highlightGridRange(dragUser, dragStartD, dragEndD);
    }

    function handleTouchMove(e) {
        if (!isDragging) return;
        e.preventDefault(); // Prevent scrolling while selecting
        const touch = e.touches[0];
        const el = document.elementFromPoint(touch.clientX, touch.clientY);
        if(!el) return;
        const td = el.closest('td.editable-cell');
        if (!td || td.dataset.user !== dragUser) return;
        
        dragEndD = td.dataset.date;
        highlightGridRange(dragUser, dragStartD, dragEndD);
    }

    function handleDragEnd(e) {
        if (!isDragging) return;
        isDragging = false;
        
        if (dragUser && dragStartD && dragEndD) {
            let d1 = new Date(dragStartD);
            let d2 = new Date(dragEndD);
            let finalStart = d1 <= d2 ? dragStartD : dragEndD;
            let finalEnd = d1 <= d2 ? dragEndD : dragStartD;

            if(finalStart === finalEnd) {
                // If it's a single click (no drag), don't show the bulk edit modal.
                // Let the double click handler take care of single cell edit.
                clearGridSelection();
                return;
            }

            let textDesc = `Memperbarui ${dragUser} dari ${formatDate(finalStart)} s/d ${formatDate(finalEnd)}`;
            const descEl = document.getElementById('bulkEditDesc');
            if(descEl) descEl.textContent = textDesc;

            if(selectEl) selectEl.value = '';

        }

    function highlightGridRange(user, date1, date2) {
        clearGridSelection();
        let d1 = new Date(date1);
        let d2 = new Date(date2);
        let start = d1 <= d2 ? date1 : date2;
        let end = d1 <= d2 ? date2 : date1;

        const tds = document.querySelectorAll(`td.editable-cell[data-user="${user}"]`);
        tds.forEach(td => {
            const tdDate = td.dataset.date;
            if (tdDate >= start && tdDate <= end) {
                td.classList.add('drag-selected');
            }
        });
    }

    function clearGridSelection() {
        document.querySelectorAll('.drag-selected').forEach(el => el.classList.remove('drag-selected'));
    }

       if (currentUser.role !== 'admin') return;
       const td = e.target.closest('td.editable-cell');
       if (!td || td.querySelector('select')) return; // Ignore if not editable or already editing

       const username = td.dataset.user;

               <option value="agd_TM">Team Ticket Master (TM)</option>
            </optgroup>
            <optgroup label="Absensi">
               <option value="agd_C">Cuti (C)</option>
               <option value="agd_I">Izin (I)</option>
               <option value="agd_S">Sakit (S)</option>
            </optgroup>
         </select>
       `;

       const select = td.querySelector('select');
       if (select) select.focus();

       const saveInline = (ev) => {
          const val = ev.target.value;
          if (val) {
             // Hapus jadwal/agenda lama (HANYA YANG INLINE INPUT) agar input utama tidak tertimpa tak sengaja
             allSchedules = allSchedules.filter(s => !(s.picked_by === username && date >= s.start_date && date <= s.end_date && s.description === 'Inline Input'));
             allAgendas = allAgendas.filter(a => !(a.username === username && date >= a.start_date && date <= a.end_date && a.desc === 'Inline Input'));

             if (val === 'clear') {
                showToast(`✅ Data pada tanggal ${formatDate(date)} berhasil dikosongkan`);
             } else if (val.startsWith('sched_')) {
                const schedName = val.split('_')[1];
                allSchedules.push({
                   __backendId: generateId('sched'), type: 'schedule', schedule_name: schedName, description: 'Inline Input',
                   start_date: date, end_date: date, status: 'released', batch_id: '', batch_name: '', point_min: 0, point_max: 0,
                   picked_by: username, released_at: new Date().toISOString(), created_by: currentUser.username
                });
                showToast(`✅ Jadwal ${schedName} ditambahkan untuk ${username}`);
                pushNotification(username, `Jadwal ${schedName} baru saja di-assign ke Anda`);
             } else if (val.startsWith('agd_')) {
                const agdType = val.split('_')[1];
                allAgendas.push({ id: generateId('agd'), username: username, type: agdType, start_date: date, end_date: date, desc: 'Inline Input' });
                showToast(`✅ Agenda ${agdType} ditambahkan untuk ${username}`);
             }
             renderGridView(); // Re-render grid untuk menampilkan perubahan
          } else {
             // Revert ke tampilan teks jika user memilih '-' atau nge-klik di luar (blur)
             td.innerHTML = oldContent;
          }
       };

       if(select) {
           select.addEventListener('blur', saveInline);
           select.addEventListener('change', saveInline);
       }
    }

    function saveBulkEdit() {
        const selectEl = document.getElementById('bulkEditSelect');
        const val = selectEl ? selectEl.value : '';
        
        if (!val) { showToast('❌ Pilih tipe jadwal terlebih dahulu!'); return; }

        let d1 = new Date(dragStartD);
        let d2 = new Date(dragEndD);
        let finalStart = d1 <= d2 ? dragStartD : dragEndD;
        let finalEnd = d1 <= d2 ? dragEndD : dragStartD;

        // Generate array of specific dates in the selected range
        let targetDates = [];
        let curr = new Date(finalStart);
        let end = new Date(finalEnd);
        while (curr <= end) {
            const yyyy = curr.getFullYear();
            const mm = String(curr.getMonth() + 1).padStart(2, '0');
            const dd = String(curr.getDate()).padStart(2, '0');
            targetDates.push(`${yyyy}-${mm}-${dd}`);
            curr.setDate(curr.getDate() + 1);
        }

        // Hapus hanya tanggal-tanggal spesifik yang ada di rentang seleksi 
        // (yang merupakan Inline Input, karena satu Inline Input sekarang dibuat per hari)
        allSchedules = allSchedules.filter(s => {
            if (s.picked_by === dragUser && s.description === 'Inline Input') {
                if (targetDates.includes(s.start_date)) return false; // Buang data di tanggal ini
            }
            return true;
        });
        
        allAgendas = allAgendas.filter(a => {
            if (a.username === dragUser && a.desc === 'Inline Input') {
                if (targetDates.includes(a.start_date)) return false; // Buang data di tanggal ini
            }
            return true;
        });

        if (val === 'clear') {
            showToast(`✅ Area yang dipilih berhasil dikosongkan`);
        } else if (val.startsWith('sched_')) {
            const schedName = val.split('_')[1];
            // Simpan setiap hari sebagai objek individual
            targetDates.forEach(dateStr => {
                allSchedules.push({
                    __backendId: generateId('sched'), type: 'schedule', schedule_name: schedName, description: 'Inline Input',
                    start_date: dateStr, end_date: dateStr, status: 'released', batch_id: '', batch_name: '', point_min: 0, point_max: 0,
                    picked_by: dragUser, released_at: new Date().toISOString(), created_by: currentUser.username
                });
            });
            showToast(`✅ Jadwal ${schedName} ditambahkan`);
            pushNotification(dragUser, `Jadwal ${schedName} ditugaskan ke Anda`);
        } else if (val.startsWith('agd_')) {
            const agdType = val.split('_')[1];
            // Simpan setiap hari sebagai objek individual
            targetDates.forEach(dateStr => {
                allAgendas.push({ id: generateId('agd'), username: dragUser, type: agdType, start_date: dateStr, end_date: dateStr, desc: 'Inline Input' });
            });
            showToast(`✅ Agenda ${agdType} ditambahkan`);
        }

        const modal = document.getElementById('bulkEditModal');
        if(modal) modal.classList.add('hidden');
        renderGridView();
    }

    // --- Analytics Tab Logic ---
    function renderAnalytics() {
      // 1. Calculate Analytics Data
      const teamUsers = allUsers.filter(u => u.role === 'team');
      let stats = teamUsers.map(u => {
         return {
            username: u.username,
            point: u.point,
            totalSched: 0,
            totalImpl: 0,
            totalMaint: 0,
            totalAbsence: 0
         };
      });

      let implCount = 0, maintCount = 0, otherCount = 0, schedCount = 0;

      // Count Schedules
      allSchedules.filter(s => s.status === 'released' || s.status === 'picked').forEach(s => {
         schedCount++;
         let statObj = stats.find(st => st.username === s.picked_by);
         if(statObj) statObj.totalSched++;
      });

      // Count Agendas
      allAgendas.forEach(a => {
         let statObj = stats.find(st => st.username === a.username);
         if (a.type.startsWith('I.')) { implCount++; if(statObj) statObj.totalImpl++; }
         else if (a.type.startsWith('M.')) { maintCount++; if(statObj) statObj.totalMaint++; }
         else if (['C','I','S'].includes(a.type)) { if(statObj) statObj.totalAbsence++; }
         else { otherCount++; }
      });

      // 2. Render Top Performer (Sort by points, take top 5)
      const topPerformers = [...stats].sort((a,b) => b.point - a.point).slice(0, 5);
      const topContainer = document.getElementById('topPerformerList');
      if (topContainer) {
          topContainer.innerHTML = topPerformers.map((p, index) => `
             <div class="flex justify-between items-center bg-gray-50 dark:bg-slate-800 p-3 rounded-lg border border-gray-100 dark:border-slate-700">
                <div class="flex items-center gap-3">
                   <span class="font-black text-lg ${index === 0 ? 'text-yellow-500' : index === 1 ? 'text-gray-400' : index===2 ? 'text-amber-600' : 'text-gray-500'}">#${index+1}</span>
                   <span class="font-bold text-gray-800 dark:text-gray-200">${p.username}</span>
                </div>
                <span class="bg-blue-100 text-blue-800 dark:bg-blue-900 dark:text-blue-200 font-bold px-2 py-1 rounded text-sm">${p.point} pts</span>
             </div>
          `).join('');
      }

      // 3. Render Chart.js
      const canvasEl = document.getElementById('taskDistributionChart');
      if(canvasEl) {
          const ctx = canvasEl.getContext('2d');
          if(chartInstance) chartInstance.destroy();

          const isDark = document.documentElement.classList.contains('dark');
          const textColor = isDark ? '#e2e8f0' : '#1f2937';

          chartInstance = new Chart(ctx, {
             type: 'doughnut',
             data: {
                labels: ['Schedules (D/MD/DT)', 'Project Implementation', 'Maintenance', 'Internal/Others'],
                datasets: [{
                   data: [schedCount, implCount, maintCount, otherCount],
                   backgroundColor: ['#3b82f6', '#6366f1', '#10b981', '#a855f7'],
                   borderWidth: 0,
                   hoverOffset: 4
                }]
             },
             options: {
                responsive: true,
                maintainAspectRatio: false,
                plugins: {
                   legend: { position: 'right', labels: { color: textColor, font: {family: 'sans-serif'} } }
                },
                cutout: '65%'
             }
          });
      }

      // 4. Render Table Recap
      window.currentStats = stats; // Save for search filter
      renderAnalyticsTable();
    }

    function renderAnalyticsTable() {
       if(!window.currentStats) return;
       const searchInput = document.getElementById('searchAnalyticsUser');
       const searchTerm = searchInput ? searchInput.value.toLowerCase() : '';
       let filtered = window.currentStats;
       if(searchTerm) filtered = filtered.filter(s => s.username.toLowerCase().includes(searchTerm));

       const tbody = document.getElementById('recapTableBody');
       if(tbody) {
           tbody.innerHTML = filtered.sort((a,b)=>a.username.localeCompare(b.username)).map(s => `
             <tr class="hover:bg-gray-50 dark:hover:bg-slate-800/50">
                <td class="px-4 py-3 font-bold">${s.username}</td>
                <td class="px-4 py-3 font-semibold text-blue-600 dark:text-blue-400">${s.point}</td>
                <td class="px-4 py-3">${s.totalSched} Hari</td>
                <td class="px-4 py-3">${s.totalImpl} Tugas</td>
                <td class="px-4 py-3">${s.totalMaint} Tugas</td>
                <td class="px-4 py-3 text-red-500 font-medium">${s.totalAbsence} Hari</td>
             </tr>
           `).join('');
       }
    }

      return date.toLocaleDateString('en-GB', { day: '2-digit', month: 'short', year: 'numeric' });
